const byte LED[] = {LED1,LED2,LED3,LED4};
const byte BTN[] = {BT1,BT2,BT3,BT4};

void setup() {
  for(int i=0 ; i<4 ; i++){
    pinMode(LED[i],OUTPUT);
    digitalWrite(LED[i],LOW);
  }
  for(int i=0 ; i<4 ; i++){
    pinMode(BTN[i],INPUT);
  }
}

void loop() {
    for(int i=0 ; i<4 ; i++){
      if(!digitalRead(BTN[i])){
        for(int j=0 ; j<4 ; j++)
          digitalWrite(LED[i],HIGH);
          break;
      }
    }
    delay(20);
}
