# 班級人數(建構子練習)
```
package Ksu;

public class A {
	static String a1;
	static int a2;
	
	public A() {
		a1 = "無班級";
		a2 = 0;
	}
	
	public A(String aa1,int aa2) {
		a1 = aa1;
		a2 = aa2;
	}
	
	public static void methodA() {
		System.out.printf("%s,%d\n",a1,a2);
	}
}
```
```
package Ksu;

public class B {

	public static void main(String[] args) {
		A obj1 = new A();
		obj1.methodA();
		A obj2 = new A("資工一A",40);
		obj2.methodA();
	}
}
```
