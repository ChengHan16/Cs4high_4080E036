```
from flask import Flask
app=Flask(__name__) #__name__ 代表目前執行模組

@app.route("/") #函式的裝飾 (Decorator): 以函式為基礎，提供附加的功能
def home():
    return "Hello Flask 2" 

if __name__=="__main__": #如果以主程式執行
    app.run() #立刻啟動伺服器
```
